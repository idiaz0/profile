data = pd.read_csv('/content/drive/MyDrive/Colab Notebooks/car_data.csv')
data.columns=['buying', 'maint', 'doors', 'persons', 'lug_boot', 'safety', 'class']
features = ['buying', 'maint', 'doors', 'persons', 'lug_boot', 'safety']

buyingNumber={'high': 3, 'med': 2, 'low': 1, 'vhigh': 0}
maintNumber={'high': 3, 'med': 2, 'low': 1, 'vhigh': 0}
doorsNumber={'5more': 5, '4': 4, '3': 3, '2': 2}
personsNumber={'4': 4, 'more': 5, '2': 2}
lug_bootNumber={'big': 3, 'med': 2, 'small': 1}
safetyNumber={'high': 4, 'med': 5, 'low': 2}
classNumber={'unacc': 1, 'acc': 2, 'good': 3, 'vgood': 4}

data['buying'] = data['buying'].map(buyingNumber)
data['maint']=data['maint'].map(maintNumber)
data['doors'] = data['doors'].map(doorsNumber)
data['persons']=data['persons'].map(personsNumber)
data['lug_boot']=data['lug_boot'].map(lug_bootNumber)
data['safety']=data['safety'].map(safetyNumber)
data['class']=data['class'].map(classNumber)
