pc_comps = ['PC1','PC2', 'PC3','PC4']
data_pca = pd.DataFrame(np.round(pca.components_[:4],120),index=pc_comps,columns=data_scaled.columns)

    if val <= -0.20:

    elif val >= 0.20:
        return 'background: skyblue'
